cmake_minimum_required(VERSION 3.30)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

set(CMAKE_EXPERIMENTAL_CXX_IMPORT_STD
    "d0edc3af-4c50-42ea-a356-e2862fe7a444")

project(beginning_cpp)

file(GLOB_RECURSE CHAPTERS CONFIGURE_DEPENDS "chapter*/*.cpp")

include_directories(./)

foreach(src ${CHAPTERS})
    file(RELATIVE_PATH rel ${CMAKE_SOURCE_DIR} ${src})
    get_filename_component(dir ${rel} DIRECTORY)
    get_filename_component(name ${src} NAME_WE)
    string(REPLACE "/" "_" dir_flat ${dir})
    set(target "${dir_flat}_${name}")


    if(name MATCHES "^m_")
        continue()
    endif()

    get_filename_component(src_dir ${src} DIRECTORY)
    file(GLOB impl "${src_dir}/m_*.cpp")

    add_executable(${target} ${src} ${impl})

    if(dir MATCHES "chapter11")
        set_target_properties(${target} PROPERTIES CXX_MODULE_STD ON)

        get_filename_component(src_dir ${src} DIRECTORY)
        file(GLOB cppm "${src_dir}/*.cppm")

        if(cppm)
            target_sources(${target}
                PUBLIC FILE_SET my_modules TYPE CXX_MODULES
                BASE_DIRS ${src_dir}
                FILES ${cppm}
            )
        endif()
    endif()
endforeach()
